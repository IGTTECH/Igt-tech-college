rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /posts/{postId} {
      // Anyone can read posts
      allow read: if true;

      // Only specific admin UIDs can create, update, delete
      allow create, update, delete: if request.auth != null
        && request.auth.uid in [
          "ADMIN_UID_1",
          "ADMIN_UID_2"
        ];
    }
  }
}
